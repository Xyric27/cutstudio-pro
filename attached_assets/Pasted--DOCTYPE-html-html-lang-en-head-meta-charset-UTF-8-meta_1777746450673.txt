async function doLogin(){
  const email=document.getElementById('login-email').value.trim().toLowerCase();
  const pass=document.getElementById('login-pass').value;
  const errDiv=document.getElementById('login-err');

  if(!email||!pass){errDiv.textContent='Please enter email & password';errDiv.style.display='block';return}

  if(isProductionMode&&document.getElementById('cfg-apikey').value){
    await initFirebaseLogin(email,pass);return;
  }

  const user=DEMO_DATA.users.find(u=>u.email===email&&u.password===pass);
  if(!user){
    errDiv.innerHTML='Invalid credentials!<br><small style="color:rgba(255,255,255,0.4)">Try: admin@cutstudio.com / admin123</small>';
    errDiv.style.display='block';return;
  }
  currentUser=user;
  document.getElementById('login-screen').style.display='none';
  launchApp();
}

async function initFirebaseLogin(email,pass){
  try{
    if(!firebase.apps.length) firebase.initializeApp({
      apiKey:document.getElementById('cfg-apikey').value.trim(),
      projectId:document.getElementById('cfg-projectid').value.trim()
    });
    auth=firebase.auth();db=firebase.firestore();
    const snap=await db.collection('users').where('email','==',email).get();
    if(snap.empty){showToast('User not found in Firebase','error');return;}
    currentUser={uid:snap.docs[0].id,...snap.docs[0].data()};
    document.getElementById('login-screen').style.display='none';
    launchApp();showToast('✅ Connected to Firebase!','success');
  }catch(e){showToast('Firebase Error: '+e.message,'error')}
}

  showHome();
}

    try{
      new Razorpay({
        key:razorpayKey,amount:amt*100,currency:'INR',
        name:'CutStudio Pro',description:`Project #${pid}`,
        handler:function(r){
          markPaid(pid);
          body.innerHTML=`<div class="success-wrap"><div class="success-ring">✓</div><div class="success-title">PAYMENT SUCCESS!</div><div class="success-sub">ID: ${r.razorpay_payment_id}</div><button class="btn-dl" onclick="closeModal('pay-modal');openDetail('${pid}')">↓ Download Video</button></div>`;

        },
        prefill:{name:currentUser.name,email:currentUser.email},
        theme:{color:'#e8a020'}
      }).open();
      closeModal('pay-modal');
    }catch(e){simulatePay(pid,body);}
  }else{
    setTimeout(()=>{
      markPaid(pid);
      body.innerHTML=`<div class="success-wrap"><div class="success-ring">✓</div><div class="success-title">PAYMENT SUCCESS!</div><div class="success-sub">(Demo simulation complete)</div><button class="btn-dl" onclick="closeModal('pay-modal');openDetail('${pid}')">↓ View Project</button></div>`;
      showToast('Demo payment processed! ✓','success');
    },2200);
  }
}

function simulatePay(pid,body){
  markPaid(pid);
  body.innerHTML=`<div class="success-wrap"><div class="success-ring">✓</div><div class="success-title">SUCCESS!</div><div class="success-sub">Demo complete</div><button class="btn-dl" onclick="closeModal('pay-modal');openDetail('${pid}')">↓ View Project</button></div>`;
}

async function markPaid(id){
  const p=DEMO_DATA.projects.find(x=>x.id===id);
  if(p){p.status='paid';p.paidAt=new Date().toISOString();saveDemoData();}
  if(currentUser.role==='admin')loadAdminData();else loadClientProjects();
  showToast('Status updated: PAID ✓','success');
}

async function saveProject(){
  const d={
    id:'p'+Date.now(),
    title:document.getElementById('f-title').value.trim(),
    clientEmail:document.getElementById('f-client').value.trim().toLowerCase(),
    price:parseInt(document.getElementById('f-price').value)||0,
    duration:document.getElementById('f-duration').value.trim(),
    previewUrl:document.getElementById('f-preview').value.trim(),
    finalUrl:document.getElementById('f-finalUrl').value.trim(),
    desc:document.getElementById('f-desc').value.trim(),
    status:'preview',
    createdAt:new Date().toISOString()
  };
  if(!d.title||!d.clientEmail||!d.price){showToast('Fill in: Title, Email, Price','error');return;}
  if(isProductionMode&&db){await db.collection('projects').add(d);showToast('Saved to Firebase!','success');}
  else{DEMO_DATA.projects.push(d);saveDemoData();showToast('Project created! (Demo)','success');}
  closeModal('add-project-modal');loadAdminData();
}

async function deleteProject(id){
  if(!confirm('Delete this project?'))return;
  if(isProductionMode&&db)await db.collection('projects').doc(id).delete();
  else DEMO_DATA.projects=DEMO_DATA.projects.filter(p=>p.id!==id);
  saveDemoData();loadAdminData();showToast('Project deleted','success');
}

async function saveClient(){
  const d={
    uid:'u'+Date.now(),
    name:document.getElementById('c-name').value.trim(),
    email:document.getElementById('c-email').value.trim().toLowerCase(),
    phone:document.getElementById('c-phone').value.trim(),
    role:'client'
  };
  const pass=document.getElementById('c-pass').value.trim();
  if(!d.name||!d.email||!pass){showToast('Name, email & password required','error');return;}
  if(DEMO_DATA.users.find(u=>u.email===d.email)){showToast('Email already exists!','error');return;}
  d.password=pass;
  if(isProductionMode&&db){await db.collection('users').add(d);showToast('Added to Firebase!','success');}
  else{DEMO_DATA.users.push(d);saveDemoData();showToast('Client created! (Demo)','success');}
  closeModal('add-client-modal');renderClientsTable();renderAdminStats();
}

async function deleteClient(uid){
  if(!confirm('Remove this client?'))return;
  if(isProductionMode&&db)await db.collection('users').doc(uid).delete();
  else DEMO_DATA.users=DEMO_DATA.users.filter(u=>u.uid!==uid);
  saveDemoData();renderClientsTable();renderAdminStats();showToast('Client removed','success');
}

function showToast(msg,type='success'){
  const t=document.createElement('div');
  t.className=`toast ${type}`;
  t.innerHTML=`<div class="toast-dot"></div><span>${msg}</span>`;
  document.body.appendChild(t);
  setTimeout(()=>{
    t.style.animation='toastOut .35s cubic-bezier(0.23,1,0.32,1) forwards';
    setTimeout(()=>t.remove(),350);
  },3200);
}
